FROM ubuntu:22.04

# Prevent interactive prompts during package installation
ENV DEBIAN_FRONTEND=noninteractive

WORKDIR /app

# Install system dependencies
RUN apt-get update && apt-get install -y \
    wget \
    bzip2 \
    unzip \
    libfreetype6 \
    libgl1-mesa-glx \
    libglu1-mesa \
    libxi6 \
    libxrender1 \
    libxkbcommon0 \
    libgomp1 \
    libgeos-dev \
    python3.11 \
    python3-pip \
    xvfb \
    && rm -rf /var/lib/apt/lists/*

# Download and install Blender 4.2.0 (latest stable)
RUN wget -q https://download.blender.org/release/Blender4.2/blender-4.2.0-linux-x64.tar.xz && \
    tar -xf blender-4.2.0-linux-x64.tar.xz && \
    mv blender-4.2.0-linux-x64 /opt/blender && \
    rm blender-4.2.0-linux-x64.tar.xz && \
    ln -s /opt/blender/blender /usr/local/bin/blender

# Install latest BlenderBIM addon (250201 - Feb 2025)
RUN wget -q https://github.com/IfcOpenShell/IfcOpenShell/releases/download/blenderbim-250201/blenderbim-250201-py311-linux.zip && \
    mkdir -p /root/.config/blender/4.2/scripts/addons && \
    unzip -q blenderbim-250201-py311-linux.zip -d /root/.config/blender/4.2/scripts/addons/ && \
    rm blenderbim-250201-py311-linux.zip && \
    rm -rf /root/.config/blender/4.2/scripts/addons/blenderbim/libs/site/packages/shapely && \
    rm -rf /root/.config/blender/4.2/scripts/addons/blenderbim/libs/site/packages/lxml

# Install required dependencies into Blender's Python environment (required by BlenderBIM)
RUN /opt/blender/4.2/python/bin/python3.11 -m ensurepip && \
    /opt/blender/4.2/python/bin/python3.11 -m pip install --upgrade pip && \
    /opt/blender/4.2/python/bin/python3.11 -m pip install shapely==2.0.2 lxml==5.1.0

# Enable BlenderBIM addon
RUN blender --background --python-expr "import bpy; bpy.ops.preferences.addon_enable(module='blenderbim'); bpy.ops.wm.save_userpref()"

# Install Python dependencies
COPY requirements.txt .
RUN pip3 install --no-cache-dir -r requirements.txt

# Copy application files
COPY main.py .
COPY blender_generator.py .
COPY start.sh .

RUN chmod +x start.sh

EXPOSE ${PORT}

CMD ["./start.sh"]
